// Contract Addresses const stakingManagerAddr =
"0x2F9E497Eb3f11c7c286874B08b543e8f2c300f45"; const
referralRegistryAddr =
"0x2a359Fbbc7269313d81d4fB6d0Bd1f394671186d"; let signer,
provider; async function connect() {
if(!window.ethereum) return alert("Open in MetaMask!");
const accounts = await window.ethereum.request({method:
'eth_requestAccounts'}); provider = new
ethers.providers.Web3Provider(window.ethereum); signer =
provider.getSigner();
document.getElementById('walletDisplay').innerText = accounts[0];
document.getElementById('connectBtn').innerText = "Connected";
} async function stake() { const amt =
document.getElementById('auraAmt').value; const usdt =
document.getElementById('usdtAmt').value; const contract =
new ethers.Contract(stakingManagerAddr, ["function
stake(uint256,uint256)"], signer); const tx = await
contract.stake(amt, usdt); alert("Mining Transaction: " +
tx.hash); } async function generateRef() { const
contract = new ethers.Contract(referralRegistryAddr, ["function
createMyReferralCode()"], signer); const tx = await
contract.createMyReferralCode(); alert("Referral created!");
} async function register() { const code =
document.getElementById('refCodeInput').value; const
contract = new ethers.Contract(referralRegistryAddr, ["function
registerUserWithCode(bytes32)"], signer); const tx = await
contract.registerUserWithCode(code); alert("Registered!");
} document.getElementById('connectBtn').onclick = connect;
document.getElementById('stakeBtn').onclick = stake;
document.getElementById('genRefBtn').onclick = generateRef;
document.getElementById('regRefBtn').onclick = register; </script>
